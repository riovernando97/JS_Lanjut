CLOSURE

Closure adalah fitur khusus dalam JavaScript di mana sebuah fungsi dapat "mengingat" dan mengakses scope di mana ia didefinisikan, bahkan setelah fungsi tersebut dieksekusi.

Contoh 1 :

      function init() {
        let nama = 'Rio'; // local variable
        function tampilNama() {
          // inner function (*closure)
          console.log(nama); // akses ke parent variable
        }
        tampilNama();
      }
      init();

      // bisa disederhanakan menjadi
      function person() {
        return function (nama) {
          console.log(nama);
        };
      }
      let panggilNama = person();
      panggilNama('Rio');

Output di console browser

Contoh 2 :

      // Factory Function => bikin dengan sesuai function yg lain
      function ucapkanSalam(waktu) {
        return function (nama) {
          console.log(`Halo ${nama}, selamat ${waktu} semoga harimu menyenangkan`);
        };
      }

      let selamatPagi = ucapkanSalam('Pagi');
      let selamatSiang = ucapkanSalam('Siang');
      let selamatMalam = ucapkanSalam('Malam');

      selamatPagi('Rio');
      selamatMalam('Rival');

Output di console browser

Contoh 3 :

      let add = (function () {
        let counter = 0; // Private Method
        return function () {
          // Closure
          return ++counter;
        };
      })();

      counter = 100;

      console.log(add());
      console.log(add());
      console.log(add());

Output di console browser
