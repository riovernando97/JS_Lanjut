Contoh 3

      // Object Constructor
      function operasiAritmatika(a, b) {
        this.a = a;
        this.b = b;

        // Pertambahan
        this.tambah = function () {
          return this.a + this.b;
        };

        // Perkalian
        this.kali = function () {
          return this.a * this.b;
        };

        // Pengurangan
        this.kurang = function () {
          return this.a - this.b;
        };

        // Pembagian
        this.bagi = function () {
          if (this.b === 0) {
            return 'tidak bisa membagi dengan 0';
          } else {
            return this.a / this.b;
          }
        };
      }
      let operasi = new operasiAritmatika(15, 4);
      console.log(`Hasil pertambahan = ${operasi.tambah()}`);
      console.log(`Hasil perkalian = ${operasi.kali()}`);
      console.log(`Hasil pengurangan = ${operasi.kurang()}`);
      console.log(`Hasil pembagian = ${operasi.bagi()}`);
